fun main() {

    val name = "Praveen"
    val age = 21

    println("My name is $name and I am $age years old.")

    val skills = listOf("Kotlin", "Java", "Web Development")

    printSkills(name, skills)
}

fun printSkills(userName: String, skillsList: List) {
    println("Skills of $userName are:")
    for (skill in skillsList) {
        println("- $skill")
    }
}
